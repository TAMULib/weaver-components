import { ChangeDetectionStrategy, Component, Injector, Input } from '@angular/core';
import { WvrBaseComponent } from '../shared/wvr-base.component';

/**
 * A stylable list.
 */
@Component({
  selector: 'wvr-list-component',
  templateUrl: './wvr-list.component.html',
  styleUrls: ['./wvr-list.component.scss'],
  changeDetection: ChangeDetectionStrategy.Default
})
export class WvrListComponent extends WvrBaseComponent {

  /** Specifies the display format of this list.  */
  @Input() listType = 'unordered';

  constructor(injector: Injector) {
    super(injector);
  }

}

<wvre-list-wrapper #animationRoot [ngSwitch]="true">
  <ul *ngSwitchCase="listType === 'unordered'">
    <ng-container *ngTemplateOutlet="selection"></ng-container>
  </ul>

  <ul *ngSwitchCase="listType === 'unstyled'" class="list-unstyled">
    <ng-container *ngTemplateOutlet="selection"></ng-container>
  </ul>

  <ol *ngSwitchCase="listType === 'ordered'">
    <ng-container *ngTemplateOutlet="selection"></ng-container>
  </ol>

  <dl *ngSwitchCase="listType === 'described'">
    <ng-container *ngTemplateOutlet="selection"></ng-container>
  </dl>

  <ul *ngSwitchCase="listType === 'group-ul' || listType === 'group'" class="list-group">
    <ng-container *ngTemplateOutlet="selection"></ng-container>
  </ul>

  <div *ngSwitchCase="listType === 'group-div'" class="list-group">
    <ng-container *ngTemplateOutlet="selection"></ng-container>
  </div>

  <ul *ngSwitchCase="listType === 'group-flush'" class="list-group list-group-flush">
    <ng-container *ngTemplateOutlet="selection"></ng-container>
  </ul>

  <div *ngSwitchCase="listType === 'group-custom-content'" class="list-group">
    <ng-container *ngTemplateOutlet="selection"></ng-container>
  </div>

  <div *ngSwitchDefault></div>

</wvre-list-wrapper>

<ng-template #selection>
  <ng-content select="wvre-list-item, wvr-list-item-component, template[wvr-compile]"></ng-content>
</ng-template>
